SELECT DISTINCT F.MNO 商户编号,T1.ORG_NM 分公司,ID.OPEN_ID 用户ID,
       F.DT_CTE  进件日期,A.TRAN_DT 扫码日期,A.TRAN_TM 扫码时间,A.TRAN_AMT 交易金额,
     WE.PAY_DT||ZX.PAY_DT||XM.PAY_DT|| XY.PAY_DT 支付完成日期,
     WE.PAY_TM||ZX.PAY_TM||XM.PAY_TM ||XY.PAY_TM  支付完成时间,
   (CASE WHEN A.CORG_NM='ZXWXDDBC' THEN '中信微信'
           WHEN A.CORG_NM= 'CUPOCBJ' THEN '北京全渠道' 
       WHEN A.CORG_NM= 'WXDDBC' THEN '微信'
           WHEN A.CORG_NM='XMMSDDBC' THEN '厦门民生' 
       WHEN A.CORG_NM='CUP_Z' THEN '银联'
       WHEN A.CORG_NM='XYDDBC' THEN '兴业银行'
       ELSE A.CORG_NM
       END
    )AS 通道,
    WE.TRANSACTION_ID 微信订单号,ZX.TRANSACTION_ID 中信订单号,XM.SMZFMSGID 厦门民生流水号,XY.TRANSACTION_ID 兴业银行订单号
FROM PTS.T_PTS_TRANDATA A
LEFT JOIN BAP.T_BAP_MEC_IF F ON F.IN_MNO=A.IN_MNO
LEFT JOIN UAP.T_UAP_ORG T1 ON T1.ORG_UUID=F.AGENT_BRANCH_ORG_UUID
LEFT JOIN PTS.t_pts_wechat_trandata WE ON WE.TRAN_DT=A.TRAN_DT AND WE.UUID=A.UUID     ------微信二维码支付交易流水表
LEFT JOIN PTS.T_PTS_CITIC_QR_TRANDATA  ZX ON ZX.TRAN_DT=A.TRAN_DT AND ZX.UUID=A.UUID  ------中信银行流水表
LEFT JOIN PTS.t_pts_cmbcxm_trandata XM ON XM.TRAN_DT=A.TRAN_DT AND XM.UUID=A.UUID     ------厦门民生（支付宝）流水表
LEFT JOIN pts.t_pts_xyddbc_trandata XY ON XY.TRAN_DT=A.TRAN_DT AND  XY.UUID=A.UUID   ------兴业银行（微信）二维码支付交易流水表
LEFT JOIN PTS.T_PTS_VV_OPENID ID ON ID.UUID=A.UUID AND ID.TRAN_DT=A.TRAN_DT
WHERE  A.TRAN_DT = TO_CHAR(SYSDATE-1,'YYYYMMDD')------交易日期
AND    A.TRAN_FLG IN ('N','E')
AND   A.TRAN_CD IN ('10110025','10110028')
AND    A.TRAN_STS='S'
AND    F.MNO IN ('500338802000003')
